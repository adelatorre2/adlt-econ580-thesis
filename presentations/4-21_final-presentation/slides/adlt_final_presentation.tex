%% ─── Slide 2: The question and the answer ─────────────────────────────────────
\begin{frame}{Did User Fees Change What the FDA Approves?}

  \begin{itemize}
    \item<1->
      Drug overdose deaths reached 107{,}000 in 2021.
      \cshl{Controlled substances}---DEA-scheduled drugs with abuse
      potential---are central to the epidemic.
      Regulatory decisions shape what enters the market.

    \item<2->
      \pdufahl{PDUFA (1992)} and \pdufahl{GDUFA (2012)} let the FDA
      collect industry user fees in exchange for faster review.
      Standard theory: faster review benefits high-margin drugs.
      \cshl{Controlled substances} are high-margin.

    \item<3->
      \textbf{Question:} Did \pdufahl{PDUFA} or \pdufahl{GDUFA} shift
      FDA approvals toward \cshl{controlled substances}?

    \item<4->
      \textbf{\cshl{Answer: No.}}\quad
      The hypothesis is not supported by the data.
  \end{itemize}

\end{frame}


%% ─── Slide 3: Contribution ────────────────────────────────────────────────────
\begin{frame}{What This Paper Adds}

  \begin{itemize}
    \item
      First systematic test of whether \pdufahl{PDUFA} or \pdufahl{GDUFA}
      changed the \textit{therapeutic composition} of FDA approvals.

    \item
      Two literatures---FDA user-fee effects
      (\textit{Berndt et al., 2005; Olson, 2009; Philipson et al., 2008})
      and the economics of opioid supply
      (\textit{Alpert, Powell, Pacula, 2018})---have not been joined.

    \item
      Paper provides the upstream, approval-side counterpart to downstream
      studies of \cshl{opioid market dynamics}.
  \end{itemize}

\end{frame}

%% ─── Slide 5: Empirical strategy ──────────────────────────────────────────────
\begin{frame}{Three Complementary Tests, One Answer}

  \begin{itemize}
    \item<1->
      \textbf{Interrupted time series} on annual \cshl{CS} share
      around 1992 (\pdufahl{PDUFA}) and 2012 (\pdufahl{GDUFA})

    \item<2->
      \textbf{Poisson rate model} with exposure offset---tests whether
      the conditional \cshl{CS} approval rate shifted

    \item<3->
      \textbf{Stacked diff-in-diff}---\cshl{CS} vs.\ non-\cshl{CS}
      as treatment/control; group-specific trend controls are the
      key diagnostic
  \end{itemize}

  \onslide<4->{
    \begin{block}{}
      \centering
      $\log \mathbb{E}[n_{gt}]
        = \alpha_g + \lambda_t
        + \underbrace{\delta}_{\text{coeff.\ of interest}}
          \cdot \bigl(\textcolor{cscolor}{\mathrm{CS}}_g
                      \times \mathrm{Post}_t\bigr)
        + \log\!\bigl(\mathrm{Total}_t\bigr)$
    \end{block}
    \vspace{0.3ex}
    {\footnotesize\textit{%
      NDA-only sample for \pdufahl{PDUFA}; ANDA-only for
      \pdufahl{GDUFA}. Transition years 2013--2014 excluded.}}
  }

\end{frame}

%% ─── Slide 9: What this means ─────────────────────────────────────────────────
\begin{frame}{User Fees Did Not Reshape the Composition of Approvals}

  \begin{itemize}
    \item<1->
      The standard argument---faster review benefits high-margin drugs,
      \cshl{CS} drugs are high-margin---is \textbf{not supported}.

    \item<2->
      When apparent effects appear, they trace to the
      \opioidhl{industry product cycle},
      not to \pdufahl{regulatory incentives}.

    \item<3->
      \textbf{Policy channel matters:} the pathway from regulatory
      acceleration to \cshl{CS} supply runs through
      \textbf{firm innovation decisions}, not review-time margins.
  \end{itemize}

  \onslide<3->{
    \vspace{0.8ex}
    {\small Upstream approval-side evidence complements
    Alpert, Powell \& Pacula (2018) and related downstream work.}
  }

\end{frame}


%% ─── Slide 10: Limitations and conclusion ─────────────────────────────────────
\begin{frame}{What the Design Can and Cannot Show---And What's Next}

  \begin{columns}[T]

    \begin{column}{0.47\textwidth}
      \textbf{Limitations}
      {\small
      \begin{itemize}
        \item Single national time series;\\no cross-sectional control
        \item DEA scheduling is current,\\not historical-at-approval
        \item Approval data only---\\failed applications unobserved
      \end{itemize}
      }
    \end{column}

    \begin{column}{0.47\textwidth}
      \textbf{Future work}
      {\small
      \begin{itemize}
        \item Link \cshl{CS} approval counts to\\ARCOS/IQVIA prescribing data
        \item 2014 hydrocodone reschedule\\as a cleaner natural experiment
        \item Firm/therapeutic-class variation\\to identify mechanism
        \item User-fee dose-response analysis
      \end{itemize}
      }
    \end{column}

  \end{columns}

  \vspace{1.2ex}
  \centering
  {\small\pdufahl{%
    Regulatory acceleration, on the margins of the approval pipeline,\\
    did not tilt the FDA toward \cshl{controlled substances}.}}

\end{frame}
